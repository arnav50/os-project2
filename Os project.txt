#include<stdio.h>
void firstFit(int B[],int m,int P[],int n) 
{
    int tempar[n]; 
    int i,j;
    for(i=0;i<n;i++){
    tempar[i]=-1;
    }
for(i=0;i<n;i++) 
    { 
for(j=0;j<m;j++) 
        { 
            if(B[j]>=P[i]) 
            {
               tempar[i] = j; 
                B[j] -= P[i]; 
                break;
            } 
        } 
    } 
    printf("\nProcess No.\tProcess Size\tBlock no.\n"); 
    for(i=0;i<n;i++) 
    { 
      printf(" %d\t\t%d\t\t",i+1,P[i]); 
        if (tempar[i]!=-1) 
            printf("%d\n",tempar[i]+1); 
        else
            printf( "Not Allocated\n"); 
    }
}
